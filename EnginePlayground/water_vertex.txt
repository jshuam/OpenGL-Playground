#version 400 core

in vec2 position;

out vec2 texture_coords;

uniform mat4 projection_matrix;
uniform mat4 view_matrix;
uniform mat4 model_matrix;


void main( void )
{
	gl_Position = projection_matrix * view_matrix * model_matrix * vec4( position.x, 0.0, position.y, 1.0 );
	texture_coords = vec2( position.x / 2.0 + 0.5, position.y / 2.0 + 0.5 );
}
