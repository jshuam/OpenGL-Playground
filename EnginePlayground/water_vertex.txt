#version 400 core

in vec2 position;

out vec4 clip_space;

uniform mat4 projection_matrix;
uniform mat4 view_matrix;
uniform mat4 model_matrix;


void main( void )
{
	clip_space = projection_matrix * view_matrix * model_matrix * vec4( position.x, 0.0, position.y, 1.0 );
	gl_Position = clip_space;
}
